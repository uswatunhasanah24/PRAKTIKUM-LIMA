### MENAMBAH ELEMEN LIST

`

n = ['sendok', 'garpu', 'piring', 'gelas', 'sumpit']
print(n)
A = n[0:2]
B = n[2:]
print(A)
print(B)
B.append('garpu')
print('- Menambahkan list B Dengan Nlai String : ', B)
